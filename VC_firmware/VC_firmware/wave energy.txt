void setup() 
{
 pinMode(enable_pin, OUTPUT);
 digitalWrite(enable_pin, LOW);// start with motor off
 pinMode(direction_pin, OUTPUT);
 pinMode(PWM_pin, OUTPUT);
 analogWrite(PWM_pin, 0); //start with zero speed

void loop() 
{
